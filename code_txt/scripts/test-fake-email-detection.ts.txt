import prisma from "../lib/prisma";

const FAKE_EMAIL_PATTERNS = [
  /^noemail\d+@/i,
  /^test\d+@/i,
  /^fake\d+@/i,
];

async function testFakeEmailDetection() {
  console.log("=".repeat(60));
  console.log("TEST: PHÁT HIỆN EMAIL ẢO");
  console.log("=".repeat(60) + "\n");

  const sentLogs = await prisma.emailCampaignLog.findMany({
    where: { status: "SENT" },
    select: { toEmail: true }
  });

  const sentSet = new Set(sentLogs.map(l => l.toEmail.toLowerCase()));
  console.log(`Tổng SENT emails: ${sentSet.size}`);

  let fakeCount = 0;
  const fakeEmails: string[] = [];

  for (const email of sentSet) {
    for (const pattern of FAKE_EMAIL_PATTERNS) {
      if (pattern.test(email)) {
        fakeCount++;
        fakeEmails.push(email);
        break;
      }
    }
  }

  console.log(`\n📊 Email ảo phát hiện được: ${fakeCount}`);
  
  if (fakeEmails.length > 0) {
    console.log("\nDanh sách (mẫu 20):");
    fakeEmails.slice(0, 20).forEach((e, i) => {
      console.log(`  ${i + 1}. ${e}`);
    });
    
    // Kiểm tra xem đã trong blacklist chưa
    let inBlacklist = 0;
    for (const email of fakeEmails.slice(0, 10)) {
      const bl = await prisma.emailBlacklist.findUnique({ where: { email } });
      if (bl) inBlacklist++;
    }
    console.log(`\nTrong blacklist (10 mẫu đầu): ${inBlacklist}/10`);
  }

  await prisma.$disconnect();
}

testFakeEmailDetection().catch(console.error);
